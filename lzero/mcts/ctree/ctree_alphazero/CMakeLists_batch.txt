cmake_minimum_required(VERSION 3.10)
project(mcts_alphazero_batch)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Python
find_package(Python3 COMPONENTS Interpreter Development REQUIRED)

# Find pybind11
find_package(pybind11 REQUIRED)

# Include directories
include_directories(${CMAKE_CURRENT_SOURCE_DIR})
include_directories(${Python3_INCLUDE_DIRS})

# Source files
set(SOURCE_FILES
    mcts_alphazero_batch.cpp
    node_alphazero.h
)

# Create Python module
pybind11_add_module(mcts_alphazero_batch ${SOURCE_FILES})

# Compiler options
target_compile_options(mcts_alphazero_batch PRIVATE
    -O3
    -Wall
    -Wextra
    -march=native
)

# Link Python libraries
target_link_libraries(mcts_alphazero_batch PRIVATE
    ${Python3_LIBRARIES}
)

# Set output directory
set_target_properties(mcts_alphazero_batch PROPERTIES
    LIBRARY_OUTPUT_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/build"
)

# Installation
install(TARGETS mcts_alphazero_batch
    LIBRARY DESTINATION ${CMAKE_CURRENT_SOURCE_DIR}/build
)
